63template<typename Index_, typename Data_, typename Distance_, typename HnswData_>
64class HnswSearcher final : public knncolle::Searcher<Index_, Data_, Distance_> {
65private:
66 const HnswPrebuilt<Index_, Data_, Distance_, HnswData_>& my_parent;
67
68 std::priority_queue<std::pair<HnswData_, hnswlib::labeltype> > my_queue;
69
70 static constexpr bool same_internal_data = std::is_same<Data_, HnswData_>::value;
71 std::vector<HnswData_> my_buffer;
72
73public:
74 HnswSearcher(const HnswPrebuilt<Index_, Data_, Distance_, HnswData_>& parent) : my_parent(parent) {
75 if constexpr(!same_internal_data) {
76 my_buffer.resize(my_parent.my_dim);
77 }
78 }
79
80public:
81 void search(Index_ i, Index_ k, std::vector<Index_>* output_indices, std::vector<Distance_>* output_distances) {
82 my_buffer = my_parent.my_index.template getDataByLabel<HnswData_>(i);
83 Index_ kp1 = k + 1;
84 my_queue = my_parent.my_index.searchKnn(my_buffer.data(), kp1); // +1, as it forgets to discard 'self'.
85
86 if (output_indices) {
87 output_indices->clear();
88 output_indices->reserve(kp1);
89 }
90 if (output_distances) {
91 output_distances->clear();
92 output_distances->reserve(kp1);
93 }
94
95 bool self_found = false;
96 hnswlib::labeltype icopy = i;
97 while (!my_queue.empty()) {
98 const auto& top = my_queue.top();
99 if (!self_found && top.second == icopy) {
100 self_found = true;
101 } else {
102 if (output_indices) {
103 output_indices->push_back(top.second);
104 }
105 if (output_distances) {
106 output_distances->push_back(top.first);
107 }
108 }
109 my_queue.pop();
110 }
111
112 if (output_indices) {
113 std::reverse(output_indices->begin(), output_indices->end());
114 }
115 if (output_distances) {
116 std::reverse(output_distances->begin(), output_distances->end());
117 }
118
119 // Just in case we're full of ties at duplicate points, such that 'c'
120 // is not in the set. Note that, if self_found=false, we must have at
121 // least 'K+2' points for 'c' to not be detected as its own neighbor.
122 // Thus there is no need to worry whether we are popping off a non-'c'
123 // element and then returning fewer elements than expected.
124 if (!self_found) {
125 if (output_indices) {
126 output_indices->pop_back();
127 }
128 if (output_distances) {
129 output_distances->pop_back();
130 }
131 }
132
133 if (output_distances && my_parent.my_normalize) {
134 for (auto& d : *output_distances) {
135 d = my_parent.my_normalize(d);
136 }
137 }
138 }
139
140private:
141 void search_raw(const HnswData_* query, Index_ k, std::vector<Index_>* output_indices, std::vector<Distance_>* output_distances) {
142 k = std::min(k, my_parent.my_obs);
143 my_queue = my_parent.my_index.searchKnn(query, k);
144
145 if (output_indices) {
146 output_indices->resize(k);
147 }
148 if (output_distances) {
149 output_distances->resize(k);
150 }
151
152 auto position = k;
153 while (!my_queue.empty()) {
154 const auto& top = my_queue.top();
155 --position;
156 if (output_indices) {
157 (*output_indices)[position] = top.second;
158 }
159 if (output_distances) {
160 (*output_distances)[position] = top.first;
161 }
162 my_queue.pop();
163 }
164
165 if (output_distances && my_parent.my_normalize) {
166 for (auto& d : *output_distances) {
167 d = my_parent.my_normalize(d);
168 }
169 }
170 }
171
172public:
173 void search(const Data_* query, Index_ k, std::vector<Index_>* output_indices, std::vector<Distance_>* output_distances) {
174 if constexpr(same_internal_data) {
175 my_queue = my_parent.my_index.searchKnn(query, k);
176 search_raw(query, k, output_indices, output_distances);
177 } else {
178 std::copy_n(query, my_parent.my_dim, my_buffer.begin());
179 search_raw(my_buffer.data(), k, output_indices, output_distances);
180 }
181 }
182};

185class HnswPrebuilt : public knncolle::Prebuilt<Index_, Data_, Distance_> {

187 template<class Matrix_>
188 HnswPrebuilt(const Matrix_& data, const DistanceConfig<HnswData_>& distance_config, const HnswOptions& options) :
189 my_dim(data.num_dimensions()),
190 my_obs(data.num_observations()),
191 my_space(distance_config.create(my_dim)),
192 my_normalize(distance_config.normalize),
193 my_index(my_space.get(), my_obs, options.num_links, options.ef_construction)
194 {
195 auto work = data.new_known_extractor();
196 if constexpr(std::is_same<Data_, HnswData_>::value) {
197 for (Index_ i = 0; i < my_obs; ++i) {
198 auto ptr = work->next();
199 my_index.addPoint(ptr, i);
200 }
201 } else {
202 std::vector<HnswData_> incoming(my_dim);
203 for (Index_ i = 0; i < my_obs; ++i) {
204 auto ptr = work->next();
205 std::copy_n(ptr, my_dim, incoming.begin());
206 my_index.addPoint(incoming.data(), i);
207 }
208 }
209
210 my_index.setEf(options.ef_search);
211 return;
212 }
