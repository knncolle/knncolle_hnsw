101 void search(Index_ i, Index_ k, std::vector<Index_>* output_indices, std::vector<Distance_>* output_distances) {
102 my_buffer = my_parent.my_index.template getDataByLabel<HnswData_>(i);
103 Index_ kp1 = k + 1;
104 my_queue = my_parent.my_index.searchKnn(my_buffer.data(), kp1); // +1, as it forgets to discard 'self'.
105
106 if (output_indices) {
107 output_indices->clear();
108 output_indices->reserve(kp1);
109 }
110 if (output_distances) {
111 output_distances->clear();
112 output_distances->reserve(kp1);
113 }
114
115 bool self_found = false;
116 hnswlib::labeltype icopy = i;
117 while (!my_queue.empty()) {
118 const auto& top = my_queue.top();
119 if (!self_found && top.second == icopy) {
120 self_found = true;
121 } else {
122 if (output_indices) {
123 output_indices->push_back(top.second);
124 }
125 if (output_distances) {
126 output_distances->push_back(top.first);
127 }
128 }
129 my_queue.pop();
130 }
131
132 if (output_indices) {
133 std::reverse(output_indices->begin(), output_indices->end());
134 }
135 if (output_distances) {
136 std::reverse(output_distances->begin(), output_distances->end());
137 }
138
139 // Just in case we're full of ties at duplicate points, such that 'c'
140 // is not in the set. Note that, if self_found=false, we must have at
141 // least 'K+2' points for 'c' to not be detected as its own neighbor.
142 // Thus there is no need to worry whether we are popping off a non-'c'
143 // element and then returning fewer elements than expected.
144 if (!self_found) {
145 if (output_indices) {
146 output_indices->pop_back();
147 }
148 if (output_distances) {
149 output_distances->pop_back();
150 }
151 }
152
153 if (output_distances && my_parent.my_normalize) {
154 for (auto& d : *output_distances) {
155 d = my_parent.my_normalize(d);
156 }
157 }
158 }

161 void search_raw(const HnswData_* query, Index_ k, std::vector<Index_>* output_indices, std::vector<Distance_>* output_distances) {
162 k = std::min(k, my_parent.my_obs);
163 my_queue = my_parent.my_index.searchKnn(query, k);
164
165 if (output_indices) {
166 output_indices->resize(k);
167 }
168 if (output_distances) {
169 output_distances->resize(k);
170 }
171
172 size_t position = k;
173 while (!my_queue.empty()) {
174 const auto& top = my_queue.top();
175 --position;
176 if (output_indices) {
177 (*output_indices)[position] = top.second;
178 }
179 if (output_distances) {
180 (*output_distances)[position] = top.first;
181 }
182 my_queue.pop();
183 }
184
185 if (output_distances && my_parent.my_normalize) {
186 for (auto& d : *output_distances) {
187 d = my_parent.my_normalize(d);
188 }
189 }
190 }
191
192public:
193 void search(const Data_* query, Index_ k, std::vector<Index_>* output_indices, std::vector<Distance_>* output_distances) {
194 if constexpr(same_internal_data) {
195 my_queue = my_parent.my_index.searchKnn(query, k);
196 search_raw(query, k, output_indices, output_distances);
197 } else {
198 std::copy_n(query, my_parent.my_dim, my_buffer.begin());
199 search_raw(my_buffer.data(), k, output_indices, output_distances);

221 template<class Matrix_>
222 HnswPrebuilt(const Matrix_& data, const HnswOptions<HnswData_>& options) :
223 my_dim(data.num_dimensions()),
224 my_obs(data.num_observations()),
225 my_space([&]{
226 if (options.distance_options.create) {
227 return options.distance_options.create(my_dim);
228 } else if constexpr(std::is_same<HnswData_, float>::value) {
229 return static_cast<hnswlib::SpaceInterface<HnswData_>*>(new hnswlib::L2Space(my_dim));
230 } else {
231 return static_cast<hnswlib::SpaceInterface<HnswData_>*>(new SquaredEuclideanDistance<HnswData_>(my_dim));
232 }
233 }()),
234 my_normalize([&]{
235 if (options.distance_options.normalize) {
236 return options.distance_options.normalize;
237 } else if (options.distance_options.create) {
238 return std::function<HnswData_(HnswData_)>();
239 } else {
240 return std::function<HnswData_(HnswData_)>([](HnswData_ x) -> HnswData_ { return std::sqrt(x); });
241 }
242 }()),
243 my_index(my_space.get(), my_obs, options.num_links, options.ef_construction)
244 {
245 auto work = data.new_extractor();
246 if constexpr(std::is_same<Data_, HnswData_>::value) {
247 for (Index_ i = 0; i < my_obs; ++i) {
248 auto ptr = work->next();
249 my_index.addPoint(ptr, i);
250 }
251 } else {
252 std::vector<HnswData_> incoming(my_dim);
253 for (Index_ i = 0; i < my_obs; ++i) {
254 auto ptr = work->next();
255 std::copy_n(ptr, my_dim, incoming.begin());
256 my_index.addPoint(incoming.data(), i);
257 }
258 }
259
260 my_index.setEf(options.ef_search);
261 return;
262 }
